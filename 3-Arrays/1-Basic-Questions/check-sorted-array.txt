#include <bits/stdc++.h>

using namespace std;

void input(vector < int > & nums, int size) {
  for (int index = 0; index < size; index++) {
    int element;
    cin >> element;
    nums[index] = element;
  }
}

// Time Complexity is O(N) and Space Complexity is O(1).
bool isArraySorted(vector < int > & nums, int start, int end) {
  if (start == end) return true;
  if (nums[start] > nums[start + 1]) return false;
  return isArraySorted(nums, start + 1, end);
}

int main() {
  int size;
  cin >> size;
  vector < int > nums(size, 0);
  input(nums, size);
  cout << isArraySorted(nums, 0, size - 1);
  return 0;
}
