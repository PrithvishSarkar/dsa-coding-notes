#include <bits/stdc++.h>

using namespace std;

// Time Complexity is O(N) and Space Complexity is O(1)
int largestElement(vector < int > & nums) {
  return * max_element(nums.begin(), nums.end());
}

void input(int size, vector < int > & nums) {
  while (size) {
    int element;
    cin >> element;
    nums.push_back(element);
    size--;
  }
}

int main() {
  int size;
  cin >> size;
  vector < int > nums;

  input(size, nums);

  cout << largestElement(nums);

  return 0;
}
