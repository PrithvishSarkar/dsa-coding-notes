#include <bits/stdc++.h>

using namespace std;

// Time Complexity is O(N) and Space Complexity is O(1)
int secondLargestElement(vector < int > & nums) {
  // Searching for the Largest Element
  int first = * max_element(nums.begin(), nums.end());

  // Searching for the Second Largest Element
  int second = INT_MIN;
  for (int element: nums) {
    if (element == first) continue;
    if (element > second) second = element;
  }
  return second == INT_MIN ? -1 : second;
}

void input(int size, vector < int > & nums) {
  while (size) {
    int element;
    cin >> element;
    nums.push_back(element);
    size--;
  }
}

int main() {
  int size;
  cin >> size;
  vector < int > nums;

  input(size, nums);

  cout << secondLargestElement(nums);
  return 0;
}
