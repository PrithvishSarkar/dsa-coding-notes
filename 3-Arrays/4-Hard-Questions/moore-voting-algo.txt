#include <bits/stdc++.h>

using namespace std;

void input(vector < int > & nums, int n) {
  for (int index = 0; index < n; index++) {
    int element;
    cin >> element;
    nums[index] = element;
  }
}

// Boyer-Moore Majority Voting Algorithm.
// Time Complexity is O(N) and Space Complexity is O(1).
int majorityElement(vector < int > & nums, int n) {
  int element = nums[0]; // Assume that the 1st element is the majority element.
  int count = 0; // Dummy Counting Variable.
  for (int index = 0; index < n;) {
    if (nums[index] == element) count++;
    else count--;
    if (count == 0) element = nums[index + 1];
    index++;
  }
  return element;
}

int main() {
  int size;
  cin >> size;
  vector < int > nums(size, 0);
  input(nums, size);
  cout << majorityElement(nums, size);
  return 0;
}
