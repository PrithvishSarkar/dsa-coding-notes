#include <bits/stdc++.h>

using namespace std;

void input(int rows, int cols, vector < vector < int >> & matrix) {
  while (rows) {
    int tempCols = cols;
    vector < int > tempVec;

    while (tempCols) {
      int element;
      cin >> element;
      tempVec.push_back(element);
      tempCols--;
    }

    matrix.push_back(tempVec);
    rows--;
  }
}

void output(vector < vector < int >> & matrix) {
  for (auto row: matrix) {
    for (int col: row) cout << col << " ";
    cout << endl;
  }
}

// Time Complexity is O(rows * cols) and Space Complexity is O(1).
// Using First Row and Column Marking Algorithm to limit Space Complexity.
void setZeroes(vector < vector < int >> & matrix) {
  int rows = matrix.size(), cols = matrix[0].size();

  // Boolean to figure out if First Row or Column contains Zero or not.
  bool isZeroInFirstRow = false, isZeroInFirstColumn = false;

  // Marking the Row's or Column's First Element as Zero.
  // This helps to navigate throughout the Matrix and update its value later.
  for (int row = 0; row < rows; row++) {
    for (int col = 0; col < cols; col++) {
      if (matrix[row][col] == 0) {
        if (row == 0) isZeroInFirstRow = true;
        if (col == 0) isZeroInFirstColumn = true;
        matrix[row][0] = 0;
        matrix[0][col] = 0;
      }
    }
  }

  // Update Matrix except the First Row and Column.
  for (int row = 1; row < rows; row++) {
    for (int col = 1; col < cols; col++) {
      if (matrix[row][0] == 0 || matrix[0][col] == 0) matrix[row][col] = 0;
    }
  }

  // First Row update if Zero was naturally present.
  if (isZeroInFirstRow) {
    for (int col = 0; col < cols; col++) matrix[0][col] = 0;
  }

  // First Column update if Zero was naturally present.
  if (isZeroInFirstColumn) {
    for (int row = 0; row < rows; row++) matrix[row][0] = 0;
  }
}

int main() {
  int rows, cols;
  cin >> rows >> cols;

  vector < vector < int >> matrix;

  input(rows, cols, matrix);

  cout << "Before: " << endl;
  output(matrix);


  setZeroes(matrix);

  cout << endl << "After: " << endl;
  output(matrix);

  return 0;
}
