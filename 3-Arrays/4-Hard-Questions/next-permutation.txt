#include <bits/stdc++.h>

using namespace std;

// Narayana Pandita Algorithm - Traversing Right to Left.
// Time Complexity is O(N) and Space Complexity is O(1).
void nextPermutation(vector < int > & nums) {
  int size = nums.size();
  // Step-1: Locate the pivot element - the first element which drops in value during Right to Left traversal.
  int pivot = -1;
  for (int index = size - 1; index > 0; index--) {
    if (nums[index - 1] < nums[index]) {
      pivot = index - 1;
      break;
    }
  }

  // Edge Case - No Pivot Exists
  if (pivot == -1) {
    reverse(nums.begin(), nums.end());
    return;
  }

  // Step-2: Find the first element larger than pivot element when traversing Right to Left.
  int successor = -1;
  for (int index = size - 1; index > pivot; index--) {
    if (nums[index] > nums[pivot]) {
      successor = index;
      break;
    }
  }

  // Step-3: Swap pivot element and successor element.
  swap(nums[pivot], nums[successor]);

  // Step-4: Reverse Subarray to right of pivot element.
  reverse(nums.begin() + pivot + 1, nums.end());
}

void input(int size, vector < int > & nums) {
  while (size) {
    int element;
    cin >> element;
    nums.push_back(element);
    size--;
  }
}

void output(vector < int > nums) {
  for (int value: nums) cout << value << " ";
  cout << endl;
}

int main() {
  int size;
  cin >> size;

  vector < int > nums;

  input(size, nums);
  cout << "Original Vector: " << endl;
  output(nums);

  nextPermutation(nums);
  cout << "Next Permutation Vector: " << endl;
  output(nums);

  return 0;
}
