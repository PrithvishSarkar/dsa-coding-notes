#include <bits/stdc++.h>

using namespace std;

void input(vector < int > & nums, int n) {
  for (int index = 0; index < n; index++) {
    int element;
    cin >> element;
    nums[index] = element;
  }
}

// Time Complexity is O(N) and Space Complexity is O(1).
int findMaxConsecutiveOnes(vector < int > & nums, int n) {
  int temp = 0, count = 0;
  for (int index = 0; index < n; index++) {
    if (nums[index] == 1) {
      temp++;
      count = max(temp, count);
    } else temp = 0;
  }
  return count;
}

int main() {
  int size;
  cin >> size;
  vector < int > nums(size, 0);
  input(nums, size);
  cout << findMaxConsecutiveOnes(nums, size);
  return 0;
}
