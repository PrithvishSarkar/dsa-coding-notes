#include <bits/stdc++.h>

using namespace std;

void input(vector < int > & nums, int n) {
  for (int index = 0; index < n; index++) {
    int element;
    cin >> element;
    nums[index] = element;
  }
}

// Time Complexity is O(N) and Space Complexity is O(1).
int findMaxConsecutiveOnes(vector < int > & nums) {
  int tempCounter = 0, actualCounter = 0;
  
  for (int element: nums) {
    if (element == 1) {
      tempCounter++;
      actualCounter = max(actualCounter, tempCounter);
    } else tempCounter = 0;
  }
  
  return actualCounter;
}

int main() {
  int size;
  cin >> size;
  vector < int > nums(size, 0);
  input(nums, size);
  cout << findMaxConsecutiveOnes(nums);
  return 0;
}
