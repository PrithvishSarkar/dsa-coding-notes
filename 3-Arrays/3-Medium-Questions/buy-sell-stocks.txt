#include <bits/stdc++.h>

using namespace std;

// Time Complexity is O(N) and Space Complexity is O(1).
int maxProfit(vector < int > & prices) {
  int i = 0, j = 1, size = prices.size(), profit = 0;

  // Find the minimum and subtract the maximum which is to the right of minimum.
  while (j < size && i < j) {
    int difference = prices[j] - prices[i];
    if (difference < 0) {
      i = j;
      j = i + 1;
      continue;
    }
    profit = max(profit, difference);
    j++;
  }

  return profit;
}

void input(int size, vector < int > & nums) {
  while (size) {
    int element;
    cin >> element;
    nums.push_back(element);
    size--;
  }
}

int main() {
  int size;
  cin >> size;

  vector < int > prices;

  input(size, prices);

  cout << maxProfit(prices);

  return 0;
}
