#include <bits/stdc++.h>

using namespace std;

// Time Complexity is somewhat less than O(N^2) and Space Complexity is O(1).
void transpose(vector < vector < int >> & matrix) {
  int order = matrix.size(), row = 0;

  while (row < order) {
    int col = 0;
    while (col < row) {
      swap(matrix[row][col], matrix[col][row]);
      col++;
    }
    row++;
  }
}

// Time Complexity is O(N^2) and Space Complexity is O(1).
void reverseRows(vector < vector < int >> & matrix) {
  int order = matrix.size(), row = 0;

  while (row < order) {
    reverse(matrix[row].begin(), matrix[row].end());
    row++;
  }
}

// Time Complexity is O(N^2) and Space Complexity is O(1).
void rotate(vector < vector < int >> & matrix) {
  // Step-1: Calculate the Transpose of the Matrix.
  transpose(matrix);

  // Step-2: Reverse each row of the Matrix.
  reverseRows(matrix);
}

void input(int order, vector < vector < int >> & matrix) {
  for (int row = 0; row < order; row++) {
    vector < int > temp(order);
    for (int col = 0; col < order; col++) {
      int element;
      cin >> element;
      temp[col] = element;
    }
    matrix[row] = temp;
  }
}

void output(vector < vector < int >> matrix) {
  for (auto row: matrix) {
    for (int col: row) cout << col << " ";
    cout << endl;
  }
  cout << endl;
}

int main() {
  int order;
  cin >> order;

  vector < vector < int >> matrix(order);

  input(order, matrix);
  cout << "Original Matrix:" << endl;
  output(matrix);

  rotate(matrix);
  cout << "Rotated Matrix: " << endl;
  output(matrix);

  return 0;
}
