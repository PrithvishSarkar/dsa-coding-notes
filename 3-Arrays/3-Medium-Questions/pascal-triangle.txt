#include <bits/stdc++.h>

using namespace std;

// Time Complexity is slightly lower than O(numRows ^ 2) and Space Complexity is slightly lower than O(numRows ^ 2).
vector < vector < int >> generatePascalTriangle(int numRows) {
  vector < int > firstPascal = {1};
  vector < int > secondPascal = {1, 1};

  if (numRows == 1) return {firstPascal};
  if (numRows == 2) return {firstPascal, secondPascal};

  vector < vector < int >> result = {firstPascal, secondPascal};

  for (int row = 2; row < numRows; row++) {
    vector < int > previousPascal = result[row - 1];
    vector < int > temp(row + 1, 1);

    for (int idx = 1; idx < row; idx++) {
      temp[idx] = previousPascal[idx - 1] + previousPascal[idx];
    }

    result.push_back(temp);
  }

  return result;
}

void output(vector < vector < int >> matrix) {
  for (auto value: matrix) {
    for (int element: value) cout << element << " ";
    cout << endl;
  }
}

int main() {
  int num;
  cin >> num;

  output(generatePascalTriangle(num));

  return 0;
}
