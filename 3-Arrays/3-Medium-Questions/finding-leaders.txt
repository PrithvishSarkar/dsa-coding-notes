#include <bits/stdc++.h>

using namespace std;

// Leaders are the elements that are strictly greater then all right-side elements.
// Optimal approach is using Two Pointers.
// Time Complexity is O(N) and Space Complexity is O(M) where `M` ranges from 1 to N.
vector < int > leaders(vector < int > & nums) {
  int size = nums.size();
  if (size == 0) return {};

  vector < int > result;

  int i = 0, j = 1;
  while (i < j && j < size) {
    if (nums[i] <= nums[j]) {
      i++;
      j = i + 1;
      continue;
    }
    j++;
    if (j == size) {
      result.push_back(nums[i]);
      i++;
      j = i + 1;
    }
  }

  result.push_back(nums[size - 1]); // Last element is always a leader.

  return result;
}

void input(int size, vector < int > & nums) {
  while (size) {
    int element;
    cin >> element;
    nums.push_back(element);
    size--;
  }
}

void output(vector < int > nums) {
  for (int element: nums) cout << element << " ";
  cout << endl;
}

int main() {
  int size;
  cin >> size;

  vector < int > nums;

  input(size, nums);

  output(leaders(nums));

  return 0;
}
