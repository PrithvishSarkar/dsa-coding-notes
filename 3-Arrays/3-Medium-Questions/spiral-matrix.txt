#include <bits/stdc++.h>

using namespace std;

// Using Two Pointers for both Rows and Columns.
// Time Complexity is O(rows * cols) and Space Complexity is O(rows * cols).
vector < int > spiralOrder(vector < vector < int >> & matrix) {
  vector < int > result;

  int rows = matrix.size(), cols = matrix[0].size();
  int rowStart = 0, rowEnd = rows - 1, colStart = 0, colEnd = cols - 1;
  int iterations = rows * cols;


  while (iterations) {
    // Traversing Left to Right
    for (int col = colStart; col <= colEnd; col++) {
      result.push_back(matrix[rowStart][col]);
      iterations--;
    }
    rowStart++;
    if (iterations == 0) break;

    // Traversing Top to Bottom
    for (int row = rowStart; row <= rowEnd; row++) {
      result.push_back(matrix[row][colEnd]);
      iterations--;
    }
    colEnd--;
    if (iterations == 0) break;

    // Traversing Right to Left
    for (int col = colEnd; col >= colStart; col--) {
      result.push_back(matrix[rowEnd][col]);
      iterations--;
    }
    rowEnd--;
    if (iterations == 0) break;

    // Traversing Bottom to Top
    for (int row = rowEnd; row >= rowStart; row--) {
      result.push_back(matrix[row][colStart]);
      iterations--;
    }
    colStart++;
  }

  return result;
}

void input(int rows, int cols, vector < vector < int >> & matrix) {
  for (int row = 0; row < rows; row++) {
    vector < int > temp(cols, 0);
    for (int col = 0; col < cols; col++) {
      int element;
      cin >> element;
      temp[col] = element;
    }
    matrix[row] = temp;
  }
}

void output(vector < int > vec) {
  for (int value: vec) cout << value << " ";
  cout << endl;
}

int main() {
  int rows, cols;
  cin >> rows >> cols;

  vector < vector < int >> matrix(rows);

  input(rows, cols, matrix);

  output(spiralOrder(matrix));

  return 0;
}
