#include <bits/stdc++.h>

using namespace std;

int extractIntegerFromString(string s) {
  int index = 0;

  // White Space Ignored
  while (isspace(s[index]))
    index++;

  if (s[index] != '+' && s[index] != '-' && !isdigit(s[index]))
    return 0;

  // Determining Sign
  bool isNegative = s[index] == '-' ? true : false;
  if (s[index] == '+' || s[index] == '-')
    index++;

  // Skipping Zeros
  while (s[index] == '0')
    index++;

  // Calculating the Integer
  int startIdx = index;
  while (isdigit(s[index]))
    index++;
  int endIdx = index;

  if (startIdx == endIdx)
    return 0;

  string str = s.substr(startIdx, endIdx - startIdx);
  bool condition1 =
    isNegative &&
    (str.length() > 10 || (str.length() == 10 && str >= "2147483648"));
  bool condition2 = !isNegative &&
    (str.length() > 10 || (str.length() == 10 && str >= "2147483647"));
  if (condition1)
    return INT_MIN;
  if (condition2)
    return INT_MAX;

  return (isNegative ? (-1 * stoi(str)) : stoi(str));
}

int main() {
  string s;
  cin >> s;
  cout << extractIntegerFromString(s);
  return 0;
}
