#include<bits/stdc++.h>

using namespace std;

void input(int size, stack < int > & stk) {
  while (size) {
    int element;
    cin >> element;
    stk.push(element);
    size--;
  }
}

void output(stack < int > stk) {
  while (!stk.empty()) {
    cout << stk.top() << endl;
    stk.pop();
  }
}

// Queue has One Entry Point and One Exit Point and is brilliant for Stack Reverse
// Time Complexity is O(N) and Space Complexity is O(N)
void reverseStack(stack < int > & st) {
  queue < int > q;
  while (!st.empty()) {
    q.push(st.top());
    st.pop();
  }
  while (!q.empty()) {
    st.push(q.front());
    q.pop();
  }
}

int main() {
  int size;
  cin >> size;

  stack < int > stk;

  input(size, stk);

  cout << "Before Sorting: " << endl;
  output(stk);

  reverseStack(stk);

  cout << "After Sorting: " << endl;
  output(stk);

  return 0;
}
