#include<bits/stdc++.h>

using namespace std;

const int mod = 1e9 + 7;

// Using Binary Exponentiation for Power Evaluation
// The `mod` is used to limit resulting answer in every recursion
long long power(long long base, long long exponent) {
  if (exponent == 1)
    return base;

  int opValue = exponent & 1;
  if (opValue == 0)
    return power((base % mod * base % mod) % mod, exponent >> 1) % mod;
  return (base % mod *
    power((base % mod * base % mod) % mod, exponent >> 1) % mod);
}

/*
1. There are 5 choices for Even Indices and 4 choices for Odd Indices
2. Permutation Count is `20^(n/2)` for Even 'n' or `5 * 20^(n/2)` for Odd 'n'
3. Power Function is evaluated using Binary Exponentiation
4. Time Complexity is O(logN) and Space Complexity is O(1)
*/
int countGoodNumbers(long long n) {
  if (n == 1)
    return 5;

  if (n == 2)
    return 20;

  int operationValue = n & 1; // Determines if `n` is Even or Odd
  if (operationValue == 0)
    return int(power(20, n >> 1) % mod);

  return int((power(20, n >> 1) % mod * 5) % mod);
}

int main() {
  long long n; // Length of Number String
  cin >> n;
  cout << countGoodNumbers(n) << endl;
  return 0;
}
