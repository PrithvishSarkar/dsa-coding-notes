#include <bits/stdc++.h>

using namespace std;

void input(int size, vector < int > & nums) {
  while (size) {
    int element;
    cin >> element;
    nums.push_back(element);
    size--;
  }
}

void output(vector < int > & nums) {
  for (int element: nums) cout << element << endl;
}

// Time Complexity is O(N^2) and Space Complexity is O(1).
void bubbleSort(vector < int > & nums) {
  static int end = nums.size() - 1;

  if (end == 0) return;

  for (int index = 0; index < end; index++) {
    if (nums[index] > nums[index + 1]) swap(nums[index], nums[index + 1]);
  }
  end--;
  bubbleSort(nums);
}

int main() {
  int size;
  cin >> size;
  vector < int > nums;

  input(size, nums);

  cout << "Before Sorting: " << endl;
  output(nums);

  bubbleSort(nums);

  cout << "After Sorting: " << endl;
  output(nums);
  return 0;
}
