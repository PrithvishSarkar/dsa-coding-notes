#include <bits/stdc++.h>

using namespace std;

void input(int size, vector < int > & nums) {
  while (size) {
    int element;
    cin >> element;
    nums.push_back(element);
    size--;
  }
}

void output(vector < int > & nums) {
  for (int element: nums) cout << element << endl;
}

// Time Complexity is O(N^2) and Space Complexity is O(1).
void selectionSort(vector < int > & nums) {
  static int start = 0;
  if (start == nums.size() - 1) return;
  for (int index = nums.size() - 1; index > start; index--) {
    if (nums[index] < nums[index - 1]) swap(nums[index], nums[index - 1]);
  }
  start++;
  selectionSort(nums);
}

int main() {
  int size;
  cin >> size;
  vector < int > nums;

  input(size, nums);

  cout << "Before Sorting: " << endl;
  output(nums);

  selectionSort(nums);

  cout << "After Sorting: " << endl;
  output(nums);
  return 0;
}
