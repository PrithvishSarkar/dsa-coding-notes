#include <bits/stdc++.h>

using namespace std;

void input(int size, vector < int > & nums) {
  while (size) {
    int element;
    cin >> element;
    nums.push_back(element);
    size--;
  }
}

void output(vector < int > & nums) {
  for (int element: nums) cout << element << endl;
}

// Shifts Elements to Right
void shiftAndPlace(int i, int j, vector < int > & nums) {
  int temp = nums[i];
  for (int k = (i - 1); k > j; k--) nums[k + 1] = nums[k];
  nums[j + 1] = temp;
}

// Time Complexity is O(N^2) and Space Complexity is O(1)
void insertionSort(vector < int > & nums) {
  static int start = 1;

  if (start == nums.size()) return;

  int index = start - 1;
  while (index >= 0) {
    if (nums[index] > nums[start]) index--;
    else break;
  }

  shiftAndPlace(start, index, nums);

  start++;

  insertionSort(nums);
}

int main() {
  int size;
  cin >> size;
  vector < int > nums;

  input(size, nums);

  cout << "Before Sorting: " << endl;
  output(nums);

  insertionSort(nums);

  cout << "After Sorting: " << endl;
  output(nums);

  return 0;
}
