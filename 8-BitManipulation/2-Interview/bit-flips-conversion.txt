#include <bits/stdc++.h>

using namespace std;

// Time Complexity is O(N) and Space Complexity is O(1).
int minBitFlips(int start, int goal) {
  // Take the XOR and count the set bits.
  int num = start ^ goal;
  int count = 0;
  while (num) {
    int temp = num;
    int bit = temp & 1;
    if (bit) count++;
    num = num >> 1;
  }
  return count;
}

int main() {
  int start, goal;
  cin >> start >> goal;
  cout << minBitFlips(start, goal);
  return 0;
}
