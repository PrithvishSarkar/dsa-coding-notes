#include <bits/stdc++.h>

using namespace std;

// Time Complexity is O(r - l) and Space Complexity is O(1).
int findRangeXOR(int l, int r) {
  int xorValue = 0;
  while (l <= r) {
    xorValue ^= l;
    l++;
  }
  return xorValue;
}

int main() {
  int l, r;
  cin >> l >> r;
  cout << findRangeXOR(l, r);
  return 0;
}
