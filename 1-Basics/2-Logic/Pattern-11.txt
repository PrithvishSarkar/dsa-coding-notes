#include <bits/stdc++.h>

using namespace std;

void printPattern(int num) {
  for (int i = 1; i <= num; i++) {
    for (int j = 1; j <= i; j++) {
      int rem = (i + j + 1) % 2;
      cout << rem << " ";
    }
    cout << endl;
  }
}

int main() {
  int number;
  cin >> number;
  printPattern(number);
  return 0;
}
