#include <bits/stdc++.h>

using namespace std;

void printPattern(int num) {
  char ch = 'A';
  for (int i = 1; i <= num; i++) {
    for (int j = 1; j <= i; j++) {
      cout << char(ch + j - 1);
    }
    cout << endl;
  }
}

int main() {
  int number;
  cin >> number;
  printPattern(number);
  return 0;
}
