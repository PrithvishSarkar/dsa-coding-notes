#include <bits/stdc++.h>

using namespace std;

class Node {
  public: int data;
  Node * next;

  Node(int data) {
    this -> data = data;
    next = nullptr;
  }
};

class List {
  Node * head;
  Node * tail;

  public:
    List() {
      head = nullptr;
      tail = nullptr;
    }

  void insert(int);
  Node * getHead() {
    return head;
  };
};

void List::insert(int data) {
  Node * ptr = new Node(data);
  if (head == nullptr && tail == nullptr) {
    head = ptr;
    tail = ptr;
    return;
  }
  tail -> next = ptr;
  tail = ptr;
}

void input(List & LL, int n) {
  while (n) {
    int element;
    cin >> element;
    LL.insert(element);
    n--;
  }
}

void print(Node * head) {
  Node * ptr = head;
  while (ptr != nullptr) {
    cout << ptr -> data << " ";
    ptr = ptr -> next;
  }
  cout << endl;
}

int findListLength(Node * head) {
  int length = 0;
  Node * ptr = head;
  while (ptr != nullptr) {
    length++;
    ptr = ptr -> next;
  }
  return length;
}

// Time Complexity is O(N) and Space Complexity is O(1).
Node * removeNodeFromEnd(Node * head, int node) {
  // Finding total length.
  int length = findListLength(head);
  node = length - node - 1;

  if (node == -1) return head -> next;

  Node * ptr = head;
  while (node) {
    node--;
    ptr = ptr -> next;
  }

  ptr -> next = ptr -> next -> next;
  return head;
}

int main() {
  int size, node;
  cin >> size >> node;
  List LL;
  input(LL, size);

  print(LL.getHead()); // Before Deletion.
  Node * head = removeNodeFromEnd(LL.getHead(), node);
  print(head); // After Deletion.
  return 0;
}
