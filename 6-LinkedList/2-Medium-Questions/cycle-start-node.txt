#include <bits/stdc++.h>
using namespace std;

class Node {
  public:
  int data;
  Node * next;
  
  Node(int data) {
    this->data = data;
    next = nullptr;
  }
};

class List {
  public:
  Node * head;
  Node * tail;
  
  List() {
    head = nullptr;
    tail = nullptr;
  }
  
  void insert(int);
};

void List::insert(int data) {
  Node * ptr = new Node(data);
  if (head == nullptr && tail == nullptr) {
    head = ptr;
    tail = ptr;
    return;
  }
  tail -> next = ptr;
  tail = ptr;
}

void input(List & LL, int size, int pos) {
  while (size) {
    int element;
    cin >> element;
    LL.insert(element);
    size--;
  }
  
  Node * temp = LL.head;
  while (pos > 0) {
    pos--;
    temp = temp -> next;
  }
  LL.tail -> next = pos == -1 ? NULL : temp;
}

Node * cycleNode(Node * head) {
  Node * fast = head;
  Node * slow = head;
  while (fast != nullptr && fast -> next != nullptr) {
    fast = fast -> next -> next;
    slow = slow -> next;
    if (fast == slow) return slow;
  }
  return nullptr;
}

// Time Complexity is O(N) and Space Complexity is O(1).
Node * loopStartNode(Node * head) {
  Node * ptr1 = cycleNode(head);
  if (ptr1 == nullptr) return nullptr;
  
  Node * ptr2 = head;
  while (ptr1 != ptr2) {
    ptr1 = ptr1 -> next;
    ptr2 = ptr2 -> next;
  }
  return ptr1;
}

int main() {
	int size, loopNodeIndex;
	cin >> size >> loopNodeIndex;
	List LL;
	input(LL, size, loopNodeIndex);
	
	Node * ptr = loopStartNode(LL.head);
	if (ptr != nullptr)	cout << ptr -> next << " " << ptr -> data << endl;
  return 0;
}
