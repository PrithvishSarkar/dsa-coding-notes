#include <bits/stdc++.h>

using namespace std;

int priority(char ch) {
  switch (ch) {
    case '^':
      return 3;
    case '*':
    case '/':
      return 2;
    case '+':
    case '-':
      return 1;
    default:
      return -1;
  }
}

/*
Steps to convert Infix to Prefix:
1. Reverse the string.
2. Swap open and close brackets.
3. Convert to Postfix (slightly modified).
4. Reverse the result.
*/

string reverse_string(string str) {
  int length = str.length();
  int start = 0, end = length - 1;
  while (start < end) swap(str[start++], str[end--]);
  return str;
}

string reverse_parantheses(string str) {
  int i = -1, j = -1, k = 0, length = str.length();
  while (k < length) {
    switch (str[k]) {
      case '(':
        i = k;
        k++;
        break;
      case ')':
        j = k;
        k++;
        break;
      default:
        k++;
        break;
    }
  }
  swap(str[i], str[j]);
  return str;
}

string infix_to_prefix(string str) {
  // Step 1 and 2
  str = reverse_parantheses(reverse_string(str));

  // Postfix (Modified) Conversion
  string result = "";
  stack < char > stk;
  int length = str.length();

  for (char ch: str) {
    if (isalnum(ch)) {
      result += ch;
      continue;
    }

    if (ch == '(') {
      stk.push(ch);
      continue;
    }

    if (ch == ')') {
      while (!stk.empty() && stk.top() != '(') {
        result += stk.top();
        stk.pop();
      }
      if (!stk.empty()) stk.pop(); // pop the '('
      continue;
    }

    while (!stk.empty() && stk.top() != '(') {
      int characterPriority = priority(ch);
      int stackTopPriority = priority(stk.top());

      bool condition1 = characterPriority < stackTopPriority && ch != '^';
      bool condition2 = characterPriority <= stackTopPriority && ch == '^';

      if (condition1 || condition2) {
        result += stk.top();
        stk.pop();
      } else break;
    }
    stk.push(ch);
  }

  while (!stk.empty()) {
    result += stk.top();
    stk.pop();
  }

  result = reverse_string(result);

  return result;
}

int main() {
  string str;
  cin >> str;
  cout << infix_to_prefix(str) << endl;
  return 0;
}
