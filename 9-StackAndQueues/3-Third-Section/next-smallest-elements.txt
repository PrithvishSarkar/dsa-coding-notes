#include<bits/stdc++.h>

using namespace std;

/*
1. To get the Next Smaller Elements, we must use Inverted Cone-Shaped Stack as a temporary container.

2. Time Complexity is O(N) and Space Complexity is O(N).
*/
vector < int > nextSmallerElements(int size, vector < int > & nums) {
  vector < int > result(size, -1);
  stack < int > stk; // Temporary Inverted Cone-Shaped Monotonic Stack

  // Vector Reverse Traversal and populating NSE Vector
  for (int index = (size - 1); index >= 0; index--) {
    int currentElement = nums[index];
    if (stk.empty()) {
      result[index] = -1;
      stk.push(currentElement);
      continue;
    }

    // Stack Top indicates Next Smaller Element to the Current Element
    while (!stk.empty() && stk.top() >= currentElement) stk.pop();

    if (stk.empty()) {
      stk.push(currentElement);
      result[index] = -1;
      continue;
    }

    result[index] = stk.top();
    stk.push(currentElement);
  }

  return result;
}

void input(int size, vector < int > & nums) {
  while (size) {
    int element;
    cin >> element;
    nums.push_back(element);
    size--;
  }
}

void output(vector < int > & nums) {
  for (int number: nums) cout << number << endl;
}

int main() {
  int size;
  cin >> size;

  vector < int > nums;

  input(size, nums);

  vector < int > result = nextSmallerElements(size, nums);

  output(result);

  return 0;
}
