#include <bits/stdc++.h>

using namespace std;

// Next Smaller Elements for each Vector element
vector < int > NSE_Indices(vector < int > & nums) {
  int size = nums.size();
  vector < int > result(size, 0);
  stack < int > stk;
  for (int i = (size - 1); i >= 0; i--) {
    while (!stk.empty() && nums[i] <= nums[stk.top()]) stk.pop();
    result[i] = stk.empty() ? size : stk.top();
    stk.push(i);
  }
  return result;
}

// Previous Smaller or Equal Elements for each Vector element
vector < int > PSEE_Indices(vector < int > & nums) {
  int size = nums.size();
  vector < int > result(size, 0);
  stack < int > stk;
  for (int i = 0; i < size; i++) {
    while (!stk.empty() && nums[i] < nums[stk.top()]) stk.pop();
    result[i] = stk.empty() ? -1 : stk.top();
    stk.push(i);
  }
  return result;
}

// Find the minimum of all the Subarrays and add them all
// Time Complexity is O(N) and Space Complexity is O(1);
int sumOfSubarrayMinimums(vector < int > & nums) {
  int size = nums.size();

  vector < int > NSE = NSE_Indices(nums);
  vector < int > PSEE = PSEE_Indices(nums);

  int total = 0, mod = int(1e9 + 7);

  for (int i = 0; i < size; i++) {
    int left = i - PSEE[i];
    int right = NSE[i] - i;
    total = (total + (left * right * 1LL * nums[i]) % mod) % mod;
  }
  return total;
}

void input(int size, vector < int > & nums) {
  while (size) {
    int element;
    cin >> element;
    nums.push_back(element);
    size--;
  }
}

int main() {
  int size;
  cin >> size;
  vector < int > nums;
  input(size, nums);
  cout << sumOfSubarrayMinimums(nums);
  return 0;
}
