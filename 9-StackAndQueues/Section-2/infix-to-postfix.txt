#include <bits/stdc++.h>

using namespace std;

// [+, -, *, /] are left associative operators 
// [^] is right associative operator
int priority(char ch) {
  switch (ch) {
    case '^':
      return 3;
    case '/':
    case '*':
      return 2;
    case '+':
    case '-':
      return 1;
    default:
      return -1;
  }
}

string infix_to_postfix(string str) {
  stack < char > stk;
  string result = "";

  for (char ch: str) {
    // Operand
    if (isalnum(ch)) { // handles a-z, A-Z, 0-9
      result += ch;
      continue;
    }

    // Opening bracket
    if (ch == '(') {
      stk.push(ch);
      continue;
    }

    // Closing bracket
    if (ch == ')') {
      while (!stk.empty() && stk.top() != '(') {
        result += stk.top();
        stk.pop();
      }
      if (!stk.empty()) stk.pop(); // pop the '('
      continue;
    }

    // Operator - pop out if it's lower priority
    while (!stk.empty() && stk.top() != '(') {
      // Left associative: pop if current priority <= stack top
      // Right associative for ^ : pop only if current priority < stack top
      int curPrio = priority(ch);
      int stackPrio = priority(stk.top());
      
      bool condition1 = curPrio <= stackPrio && ch != '^';
      bool condition2 = curPrio < stackPrio && ch == '^';

      if (condition1 || condition2) {
        result += stk.top();
        stk.pop();
      } else break;
    }
    stk.push(ch);
  }

  // Pop all remaining operators
  while (!stk.empty()) {
    result += stk.top();
    stk.pop();
  }

  return result;
}

int main() {
  string str;
  cin >> str;
  cout << infix_to_postfix(str) << endl;
  return 0;
}
