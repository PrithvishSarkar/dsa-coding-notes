#include <bits/stdc++.h>

using namespace std;

// Modified Binary Search. Time Complexity is O(logN) and Space Complexity is O(1).
bool search(vector < int > nums, int target) {
  int start = 0, end = nums.size() - 1;

  while (start <= end) {
    int mid = start + (end - start) / 2;

    if (nums[mid] == target) return true;
    
    // Edge Case if 2nd or 2nd Last Element is smaller and the targeted one.
    if (nums[mid] == nums[start] && nums[mid] == nums[end]) {
      start++;
      end--;
      continue;
    }

    if (nums[mid] >= nums[start]) {
      if (target >= nums[start] && target < nums[mid]) end = mid - 1;
      else start = mid + 1;
      continue;
    }

    if (target > nums[mid] && target <= nums[end]) start = mid + 1;
    else end = mid - 1;
  }

  return false;
}

void input(int size, vector < int > & nums) {
  while (size) {
    int element;
    cin >> element;
    nums.push_back(element);
    size--;
  }
}

int main() {
  int size;
  cin >> size;

  vector < int > nums;

  input(size, nums);

  int target;
  cin >> target;

  cout << search(nums, target);

  return 0;
}
