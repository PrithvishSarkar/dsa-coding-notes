#include <bits/stdc++.h>

using namespace std;

// Modified Binary Search. Time Complexity is O(logN) and Space Complexity is O(1).
int searchIndex(vector < int > nums, int target) {
  int start = 0, end = nums.size() - 1;

  while (start <= end) {
    int mid = start + (end - start) / 2;

    if (nums[mid] == target) return mid;

    // Left Rotated Sorted Half
    if (nums[mid] >= nums[start]) {
      if (target >= nums[start] && target < nums[mid]) end = mid - 1;
      else start = mid + 1;
      continue;
    }
    
    // Right Rotated Sorted Half
    if (target > nums[mid] && target <= nums[end]) start = mid + 1;
    else end = mid - 1;
  }

  return -1;
}

int search(vector < int > nums, int target) {
  int size = nums.size();

  if (size == 0 || (size == 1 && nums[0] != target)) return -1;

  if (size == 1 && nums[0] == target) return 0;

  return searchIndex(nums, target);
}

void input(int size, vector < int > & nums) {
  while (size) {
    int element;
    cin >> element;
    nums.push_back(element);
    size--;
  }
}

int main() {
  int size;
  cin >> size;

  vector < int > nums;

  input(size, nums);

  int target;
  cin >> target;

  cout << search(nums, target);

  return 0;
}
