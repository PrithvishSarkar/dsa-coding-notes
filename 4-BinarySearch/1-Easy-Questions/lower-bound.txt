#include <bits/stdc++.h>

using namespace std;

// Lower Bound (Sorted Array): arr[min_ele_idx] >= x
// Time Complexity is O(logN) and Space Complexity is O(1).
int search(vector < int > & nums, int target, int start, int end, int result) {
  if (start > end) return result;

  int mid = (start + end) / 2;

  if (nums[mid] < target) return search(nums, target, mid + 1, end, result);

  if (nums[mid] >= target) return search(nums, target, start, mid - 1, mid);

  return -1; // Dummy Return Statement.
}

int lowerBound(vector < int > & nums, int target) {
  return search(nums, target, 0, nums.size() - 1, nums.size());
}

void input(int size, vector < int > & nums) {
  while (size) {
    int element;
    cin >> element;
    nums.push_back(element);
    size--;
  }
}

int main() {
  int size;
  cin >> size;

  vector < int > nums;

  input(size, nums);

  int target;
  cin >> target;

  cout << lowerBound(nums, target);

  return 0;
}
