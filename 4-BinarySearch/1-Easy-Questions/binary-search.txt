#include <bits/stdc++.h>

using namespace std;

void input(vector < int > & nums, int size) {
  for (int index = 0; index < size; index++) {
    int element;
    cin >> element;
    nums[index] = element;
  }
}

// Time Complexity is O(logN) and Space Complexity is O(1).
int binarySearch(vector < int > & nums, int start, int end, int k) {
  if (start > end) return -1;
  int mid = start + (end - start) / 2;
  if (nums[mid] < k) return binarySearch(nums, mid + 1, end, k);
  else if (nums[mid] > k) return binarySearch(nums, start, mid - 1, k);
  else return mid;
}

int main() {
  int size, target;
  cin >> size >> target;
  vector < int > nums(size, 0);
  input(nums, size);
  cout << binarySearch(nums, 0, size - 1, target);
  return 0;
}
