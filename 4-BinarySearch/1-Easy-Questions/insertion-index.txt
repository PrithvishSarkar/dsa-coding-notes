#include <bits/stdc++.h>

using namespace std;

void input(vector < int > & nums, int size) {
  for (int index = 0; index < size; index++) {
    int element;
    cin >> element;
    nums[index] = element;
  }
}

void print(vector < int > & nums, int size) {
  for (int idx = 0; idx < size; idx++) cout << nums[idx] << " ";
  cout << endl;
}

// Time Complexity is O(logN) and Space Complexity is O(1).
int insertionIndex(vector < int > & nums, int n, int k) {
  if (k < nums[0]) return 0;
  else if (k > nums[n - 1]) return n;
  int low = 0, high = n - 1, idx = -1;
  while (low <= high) {
    int mid = low + (high - low) / 2;
    if (nums[mid] == k) return mid;
    else if (nums[mid] < k) low = mid + 1;
    else {
      idx = mid;
      high = mid - 1;
    }
  }
  return idx;
}

int main() {
  int size, target;
  cin >> size >> target;
  vector < int > nums(size, 0);
  input(nums, size);
  sort(nums.begin(), nums.end()); // Sorting Array in Ascending Order.
  print(nums, size);
  cout << insertionIndex(nums, size, target);
  return 0;
}
