#include <bits/stdc++.h>

using namespace std;

// Time Complexity is O(logN) and Space Complexity is O(1).
int squareRoot(int num) {
  int low = 0, high = num;
  while (low <= high) {
    int mid = (low + high) / 2;
    int squaredNumber = mid * mid;
    if (squaredNumber > num) high = mid - 1;
    else if (squaredNumber < num) low = mid + 1;
    else return mid;
  }
  return high;
}

int main() {
  int num;
  cin >> num;
  cout << squareRoot(num);
  return 0;
}
