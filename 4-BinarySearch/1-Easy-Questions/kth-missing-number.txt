#include <bits/stdc++.h>

using namespace std;

void input(vector < int > & nums, int n) {
  for (int index = 0; index < n; index++) {
    int element;
    cin >> element;
    nums[index] = element;
  }
}

// Time Complexity is O(k) and Space Complexity is O(1).
int findMissingNumber(vector < int > & nums, int n, int k) {
  int num = 1, index = 0;
  while (k) {
    if (index < n && num == nums[index]) {
      num++;
      index++;
    } else {
      k--;
      if (k == 0) break;
      num++;
    }
  }
  return num;
}

int main() {
  int size, offset;
  cin >> size >> offset;
  vector < int > nums(size, 0);
  input(nums, size);
  cout << findMissingNumber(nums, size, offset);
  return 0;
}
