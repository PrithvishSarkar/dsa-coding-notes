#include <bits/stdc++.h>

using namespace std;

void input(vector < int > & nums, int n) {
  for (int index = 0; index < n; index++) {
    int element;
    cin >> element;
    nums[index] = element;
  }
}

int highestElement(vector < int > & nums, int size) {
  int element = INT_MIN;
  for (int index = 0; index < size; index++) {
    element = max(element, nums[index]);
  }
  return element;
}
int lowestPossibleDivisor(vector < int > & nums, int n, int k) {
  double sum = 0;
  for (int index = 0; index < n; index++)
    sum += double(nums[index]) / double(k);
  sum = ceil(sum);
  return int(sum);
}
int smallestDivisor(vector < int > & nums, int n, int k) {
  if (n == k)
    return highestElement(nums, n);

  int low = lowestPossibleDivisor(nums, n, k);
  int high = highestElement(nums, n);
  while (low <= high) {
    int mid = (low + high) / 2;
    int sum = 0;
    for (int idx = 0; idx < n; idx++) {
      sum += ceil(floor(nums[idx]) / floor(mid));
    }
    if (sum <= k)
      high = mid - 1;
    else
      low = mid + 1;
  }
  return low;
}

int main() {
  int size, threshold;
  cin >> size >> threshold;
  vector < int > nums(size, 0);
  input(nums, size);
  cout << smallestDivisor(nums, size, threshold);
  return 0;
}
