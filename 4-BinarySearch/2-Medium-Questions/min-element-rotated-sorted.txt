#include <bits/stdc++.h>

using namespace std;

void input(vector < int > & nums, int n) {
  for (int index = 0; index < n; index++) {
    int element;
    cin >> element;
    nums[index] = element;
  }
}

// Time Complexity is O(logN) and Space Complexity is O(1).
int findMinimumValue(vector < int > & nums, int n) {
  int low = 0, high = n - 1, result = INT_MAX;
  while (low <= high) {
    int mid = (low + high) / 2;
    if (nums[mid] >= nums[low]) {
      result = min(result, nums[low]);
      low = mid + 1;
    } else {
      result = min(result, nums[mid]);
      high = mid - 1;
    }
  }
  return result;
}

int main() {
  int size;
  cin >> size;
  vector < int > nums(size, 0);
  input(nums, size);
  cout << findMinimumValue(nums, size);
  return 0;
}
