#include <bits/stdc++.h>

using namespace std;

void input(vector < int > & nums, int n) {
  for (int index = 0; index < n; index++) {
    int element;
    cin >> element;
    nums[index] = element;
  }
}

// Time Complexity is O(logN) and Space Complexity is O(1).
int numberOfRotations(vector < int > & nums, int n) {
  int left = 0, right = n - 1;
  while (left <= right) {
    int mid = (left + right) / 2;
    if (nums[mid] >= nums[0]) left = mid + 1;
    else {
      if (nums[mid] < nums[mid - 1]) return mid;
      else right = mid;
    }
  }
  return n;
}

int main() {
  int size;
  cin >> size;
  vector < int > nums(size, 0);
  input(nums, size);
  cout << numberOfRotations(nums, size);
  return 0;
}
