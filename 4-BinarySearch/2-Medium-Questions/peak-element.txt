#include <bits/stdc++.h>

using namespace std;

void input(vector < int > & nums, int n) {
  for (int index = 0; index < n; index++) {
    int element;
    cin >> element;
    nums[index] = element;
  }
}

// Time Complexity is O(logN) and Space Complexity is O(1).
int findPeakElementIndex(vector < int > & nums, int n) {
  int low = 0, high = n - 1;
  while (low < high) {
    int mid = low + (high - low) / 2;
    if (nums[mid] > nums[mid + 1])
      high = mid;
    else
      low = mid + 1;
  }
  return low;
}
int main() {
  int size;
  cin >> size;
  vector < int > nums(size, 0);
  input(nums, size);
  cout << findPeakElementIndex(nums, size);
  return 0;
}
