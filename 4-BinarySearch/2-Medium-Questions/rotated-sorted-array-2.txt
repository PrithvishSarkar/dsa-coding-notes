#include <bits/stdc++.h>

using namespace std;

void input(vector < int > & nums, int n) {
  for (int index = 0; index < n; index++) {
    int element;
    cin >> element;
    nums[index] = element;
  }
}

// Time Complexity is O(logN) and Space Complexity is O(1).
bool isElementPresent(vector < int > & nums, int n, int k) {
  int start = 0, end = n - 1;
  while (start <= end) {
    int mid = start + (end - start) / 2;
    if (nums[mid] == k)
      return true;
    if (nums[mid] == nums[start] && nums[mid] == nums[end]) {
      start++;
      end--;
      continue;
    }
    if (nums[start] <= nums[mid]) {
      if (k >= nums[start] && k < nums[mid])
        end = mid - 1;
      else
        start = mid + 1;
    } else {
      if (k > nums[mid] && k <= nums[end])
        start = mid + 1;
      else
        end = mid - 1;
    }
  }
  return false;
}

int main() {
  int size, target;
  cin >> size >> target;
  vector < int > nums(size, 0);
  input(nums, size);
  cout << isElementPresent(nums, size, target);
  return 0;
}
