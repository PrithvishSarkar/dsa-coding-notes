#include <bits/stdc++.h>

using namespace std;

void input(vector < int > & nums, int n) {
  for (int index = 0; index < n; index++) {
    int element;
    cin >> element;
    nums[index] = element;
  }
}

// Time Complexity is O(logN) and Space Complexity is O(1).
int findJumpIndex(vector < int > & nums, int n) {
  int low = 0, high = n - 1;
  while (low < high) {
    int mid = (low + high) / 2;
    if (nums[mid] >= nums[0]) {
      if (nums[mid] > nums[mid + 1]) return mid;
      else low = mid + 1;
    } else high = mid;
  }
  return n; // Jump Index 'n' means the array is sorted but not rotated.
}

// Time Complexity is O(logN) and Space Complexity is O(1).
int binarySearch(vector < int > & nums, int n, int start, int end, int k) {
  while (start <= end) {
    int mid = (start + end) / 2;
    if (nums[mid] == k) return mid;
    else if (nums[mid] < k) start = mid + 1;
    else end = mid - 1;
  }
  return -1;
}

int searchIndex(vector < int > & nums, int n, int k) {
  if (n == 1 && nums[0] == k) return 0;
  if (n == 1 && nums[0] != k) return -1;

  int jumpIndex = findJumpIndex(nums, n);

  // Binary Search on the entire array.
  if (jumpIndex == n) return binarySearch(nums, n, 0, n - 1, k);

  // Binary Search on part of the array.
  if (k >= nums[0] && k <= nums[jumpIndex])
    return binarySearch(nums, n, 0, jumpIndex, k);
  else if (k >= nums[jumpIndex + 1] && k <= nums[n - 1])
    return binarySearch(nums, n, jumpIndex + 1, n - 1, k);
  return -1;
}

int main() {
  int size, target;
  cin >> size >> target;
  vector < int > nums(size, 0);
  input(nums, size);
  cout << searchIndex(nums, size, target);
  return 0;
}
